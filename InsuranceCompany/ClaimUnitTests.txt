    public class ClaimServiceTests
    {
        private readonly Mock<IClaimRepository> _mockClaimRepository;
        private readonly ClaimService _claimService;

        public ClaimServiceTests()
        {
            _mockClaimRepository = new Mock<IClaimRepository>();
            _claimService = new ClaimService(_mockClaimRepository.Object);

        #region FileClaimAsync Tests

        [Fact]
        public async Task FileClaimAsync_ShouldReturnCreatedClaim_WhenSuccessfullyFiled()
        {
            int userId = 1;
            var dto = new ClaimFileDto
            {
                IssuedPolicyId = 10,
                EstimatedLossAmount = 5000.00m,
                IncidentDescription = "Water damage due to leak.",
                IncidentDate = DateTime.UtcNow.AddDays(-2)
            };

            var expectedClaim = new Claim
            {
                ClaimId = 101,
                IssuedPolicyId = dto.IssuedPolicyId,
                EstimatedLossAmount = dto.EstimatedLossAmount,
                IncidentDescription = dto.IncidentDescription,
                IncidentDate = dto.IncidentDate,
                Status = "ClaimFiled",
                FiledAt = DateTime.UtcNow
            };

            _mockClaimRepository
                .Setup(r => r.AddClaimAsync(It.IsAny<Claim>()))
                .ReturnsAsync(expectedClaim);

            var result = await _claimService.FileClaimAsync(userId, dto);

            Assert.NotNull(result);
            Assert.Equal(expectedClaim.ClaimId, result.ClaimId);
            Assert.Equal(expectedClaim.IssuedPolicyId, result.IssuedPolicyId);
            Assert.Equal(expectedClaim.Status, result.Status);
            _mockClaimRepository.Verify(r => r.AddClaimAsync(It.Is<Claim>(c => 
                c.IssuedPolicyId == dto.IssuedPolicyId && 
                c.EstimatedLossAmount == dto.EstimatedLossAmount
            )), Times.Once);
        }

        [Fact]
        public async Task FileClaimAsync_ShouldThrowException_WhenRepositoryFails()
        {
            int userId = 1;
            var dto = new ClaimFileDto { IssuedPolicyId = 10 };
            _mockClaimRepository
                .Setup(r => r.AddClaimAsync(It.IsAny<Claim>()))
                .ThrowsAsync(new Exception("Database error"));

            await Assert.ThrowsAsync<Exception>(() => _claimService.FileClaimAsync(userId, dto));
        }

        #endregion

        #region ReviewClaimAsync Tests

        [Fact]
        public async Task ReviewClaimAsync_ShouldReturnFalse_WhenClaimDoesNotExist()
        {
            int claimId = 999;
            int officerId = 5;
            var dto = new ClaimReviewDto { Status = "Approved", ApprovedSettlementAmount = 4000.00m, OfficerRemarks = "Looks good" };

            _mockClaimRepository
                .Setup(r => r.GetClaimByIdAsync(claimId))
                .ReturnsAsync((Claim?)null);

            var result = await _claimService.ReviewClaimAsync(claimId, officerId, dto);

            Assert.False(result);
            _mockClaimRepository.Verify(r => r.UpdateClaimAsync(It.IsAny<Claim>()), Times.Never);
        }

        [Theory]
        [InlineData("Approved")]
        [InlineData("Rejected")]
        public async Task ReviewClaimAsync_ShouldThrowInvalidOperationException_WhenClaimIsAlreadyFinalized(string existingStatus)
        {
            int claimId = 101;
            int officerId = 5;
            var dto = new ClaimReviewDto { Status = "Approved" };
            var existingClaim = new Claim { ClaimId = claimId, Status = existingStatus };

            _mockClaimRepository
                .Setup(r => r.GetClaimByIdAsync(claimId))
                .ReturnsAsync(existingClaim);

            var exception = await Assert.ThrowsAsync<InvalidOperationException>(() => 
                _claimService.ReviewClaimAsync(claimId, officerId, dto)
            );
            Assert.Equal("This claim has already been finalized.", exception.Message);
            _mockClaimRepository.Verify(r => r.UpdateClaimAsync(It.IsAny<Claim>()), Times.Never);
        }

        [Fact]
        public async Task ReviewClaimAsync_ShouldUpdateStatusAndReturnTrue_WhenApprovedSuccessfully()
        {
            int claimId = 101;
            int officerId = 5;
            var dto = new ClaimReviewDto 
            { 
                Status = "Approved", 
                ApprovedSettlementAmount = 3500.00m, 
                OfficerRemarks = "Approved after reviewing receipts." 
            };
            var existingClaim = new Claim { ClaimId = claimId, Status = "ClaimFiled" };

            _mockClaimRepository
                .Setup(r => r.GetClaimByIdAsync(claimId))
                .ReturnsAsync(existingClaim);

            _mockClaimRepository
                .Setup(r => r.UpdateClaimAsync(It.IsAny<Claim>()))
                .Returns(Task.CompletedTask);

            var result = await _claimService.ReviewClaimAsync(claimId, officerId, dto);

            Assert.True(result);
            Assert.Equal("Approved", existingClaim.Status);
            Assert.Equal(dto.ApprovedSettlementAmount, existingClaim.ApprovedSettlementAmount);
            Assert.Equal(dto.OfficerRemarks, existingClaim.OfficerRemarks);
            Assert.Equal(officerId, existingClaim.ReviewedByOfficerId);
            Assert.NotNull(existingClaim.UpdatedAt);
            _mockClaimRepository.Verify(r => r.UpdateClaimAsync(existingClaim), Times.Once);
        }

        [Fact]
        public async Task ReviewClaimAsync_ShouldSetSettlementToZeroAndReturnTrue_WhenRejectedSuccessfully()
        {
            int claimId = 101;
            int officerId = 5;
            var dto = new ClaimReviewDto 
            { 
                Status = "Rejected", 
                ApprovedSettlementAmount = 3500.00m, 
                OfficerRemarks = "Not covered by policy terms." 
            };
            var existingClaim = new Claim { ClaimId = claimId, Status = "ClaimFiled" };

            _mockClaimRepository
                .Setup(r => r.GetClaimByIdAsync(claimId))
                .ReturnsAsync(existingClaim);

            _mockClaimRepository
                .Setup(r => r.UpdateClaimAsync(It.IsAny<Claim>()))
                .Returns(Task.CompletedTask);

            var result = await _claimService.ReviewClaimAsync(claimId, officerId, dto);

            Assert.True(result);
            Assert.Equal("Rejected", existingClaim.Status);
            Assert.Equal(0, existingClaim.ApprovedSettlementAmount);
            Assert.Equal(dto.OfficerRemarks, existingClaim.OfficerRemarks);
            Assert.Equal(officerId, existingClaim.ReviewedByOfficerId);
            _mockClaimRepository.Verify(r => r.UpdateClaimAsync(existingClaim), Times.Once);
        }

        #endregion

        #region TrackClaimStatusAsync Tests

        [Fact]
        public async Task TrackClaimStatusAsync_ShouldReturnNull_WhenClaimDoesNotExist()
        {
            int claimId = 999;
            int userId = 1;
            _mockClaimRepository
                .Setup(r => r.GetClaimByIdAsync(claimId))
                .ReturnsAsync((Claim?)null);

            var result = await _claimService.TrackClaimStatusAsync(claimId, userId, false);

            Assert.Null(result);
        }

        [Fact]
        public async Task TrackClaimStatusAsync_ShouldReturnClaimDirectly_WhenBypassOwnershipCheckIsTrue()
        {
            int claimId = 101;
            int userId = 999;
            var claim = new Claim { ClaimId = claimId, IssuedPolicyId = 5 };

            _mockClaimRepository
                .Setup(r => r.GetClaimByIdAsync(claimId))
                .ReturnsAsync(claim);

            var result = await _claimService.TrackClaimStatusAsync(claimId, userId, true);

            Assert.NotNull(result);
            Assert.Equal(claim, result);
            _mockClaimRepository.Verify(r => r.GetClaimsByUserIdAsync(It.IsAny<int>()), Times.Never);
        }

        [Fact]
        public async Task TrackClaimStatusAsync_ShouldReturnClaim_WhenBypassIsFalseAndClaimBelongsToUser()
        {
            int claimId = 101;
            int userId = 1;
            var claim = new Claim { ClaimId = claimId, IssuedPolicyId = 5 };
            var userClaimsList = new List<Claim> { claim };

            _mockClaimRepository
                .Setup(r => r.GetClaimByIdAsync(claimId))
                .ReturnsAsync(claim);

            _mockClaimRepository
                .Setup(r => r.GetClaimsByUserIdAsync(userId))
                .ReturnsAsync(userClaimsList);

            var result = await _claimService.TrackClaimStatusAsync(claimId, userId, false);

            Assert.NotNull(result);
            Assert.Equal(claim, result);
        }

        [Fact]
        public async Task TrackClaimStatusAsync_ShouldThrowUnauthorizedAccessException_WhenBypassIsFalseAndClaimDoesNotBelongToUser()
        {
            int claimId = 101;
            int userId = 1;
            var claim = new Claim { ClaimId = claimId, IssuedPolicyId = 5 };
            var differentClaim = new Claim { ClaimId = 102, IssuedPolicyId = 5 };
            var userClaimsList = new List<Claim> { differentClaim };

            _mockClaimRepository
                .Setup(r => r.GetClaimByIdAsync(claimId))
                .ReturnsAsync(claim);

            _mockClaimRepository
                .Setup(r => r.GetClaimsByUserIdAsync(userId))
                .ReturnsAsync(userClaimsList);

            var exception = await Assert.ThrowsAsync<UnauthorizedAccessException>(() => 
                _claimService.TrackClaimStatusAsync(claimId, userId, false)
            );
            Assert.Equal("Not your claim!", exception.Message);
        }

    public class ClaimControllerTests
    {
        private readonly Mock<IClaimService> _mockClaimService;
        private readonly ClaimController _controller;

        public ClaimControllerTests()
        {
            _mockClaimService = new Mock<IClaimService>();
            _controller = new ClaimController(_mockClaimService.Object);
        }

        private void SetupUserIdentity(string nameIdentifier, params string[] roles)
        {
            var claims = new List<System.Security.Claims.Claim>
            {
                new System.Security.Claims.Claim(ClaimTypes.NameIdentifier, nameIdentifier)
            };
            foreach (var role in roles)
            {
                claims.Add(new System.Security.Claims.Claim(ClaimTypes.Role, role));
            }

            var identity = new ClaimsIdentity(claims, "TestAuthType");
            var claimsPrincipal = new ClaimsPrincipal(identity);

            _controller.ControllerContext = new ControllerContext
            {
                HttpContext = new DefaultHttpContext { User = claimsPrincipal }
            };
        }

        #region FileClaim Tests

        [Fact]
        public async Task FileClaim_ShouldReturnBadRequest_WhenModelStateIsInvalid()
        {
            _controller.ModelState.AddModelError("EstimatedLossAmount", "Estimated loss must be greater than 0");
            var dto = new ClaimFileDto();

            var result = await _controller.FileClaim(dto);

            var badRequestResult = Assert.IsType<BadRequestObjectResult>(result);
            Assert.NotNull(badRequestResult.Value);
        }

        [Fact]
        public async Task FileClaim_ShouldReturnUnauthorized_WhenUserIdClaimIsMissing()
        {
            SetupUserIdentity("");
            var dto = new ClaimFileDto { IssuedPolicyId = 10 };

            var result = await _controller.FileClaim(dto);

            Assert.IsType<UnauthorizedObjectResult>(result);
        }

        [Fact]
        public async Task FileClaim_ShouldReturnCreated_WhenClaimIsFiledSuccessfully()
        {
            SetupUserIdentity("123");
            var dto = new ClaimFileDto { IssuedPolicyId = 10, EstimatedLossAmount = 500 };
            var mockClaim = new Claim { ClaimId = 1, IssuedPolicyId = 10, EstimatedLossAmount = 500 };

            _mockClaimService
                .Setup(s => s.FileClaimAsync(123, dto))
                .ReturnsAsync(mockClaim);

            var result = await _controller.FileClaim(dto);

            var objectResult = Assert.IsType<ObjectResult>(result);
            Assert.Equal(StatusCodes.Status201Created, objectResult.StatusCode);
            Assert.Equal(mockClaim, objectResult.Value);
        }

        [Fact]
        public async Task FileClaim_ShouldReturnInternalServerError_WhenServiceThrowsException()
        {
            SetupUserIdentity("123");
            var dto = new ClaimFileDto { IssuedPolicyId = 10 };
            _mockClaimService
                .Setup(s => s.FileClaimAsync(123, dto))
                .ThrowsAsync(new Exception("Database connection failure"));

            var result = await _controller.FileClaim(dto);

            var objectResult = Assert.IsType<ObjectResult>(result);
            Assert.Equal(500, objectResult.StatusCode);
        }

        #endregion

        #region GetMyClaimsHistory Tests

        [Fact]
        public async Task GetMyClaimsHistory_ShouldReturnOkWithHistory_WhenSuccessful()
        {
            SetupUserIdentity("123");
            var mockHistory = new List<Claim> { new Claim { ClaimId = 1 }, new Claim { ClaimId = 2 } };
            _mockClaimService
                .Setup(s => s.GetCustomerClaimsHistoryAsync(123))
                .ReturnsAsync(mockHistory);

            var result = await _controller.GetMyClaimsHistory();

            var okResult = Assert.IsType<OkObjectResult>(result);
            Assert.Equal(mockHistory, okResult.Value);
        }

        #endregion

        #region GetPendingQueue Tests

        [Fact]
        public async Task GetPendingQueue_ShouldReturnOkWithFlatQueue_WhenSuccessful()
        {
            var issuedPolicy = new IssuedPolicy
            {
                PolicyNumber = "POL-999",
                InsurancePolicy = new Models.PolicyManagement.InsurancePolicy { PolicyName = "Home Insurance" }
            };
            var mockPendingList = new List<Claim>
            {
                new Claim 
                { 
                    ClaimId = 1, 
                    IssuedPolicyId = 10, 
                    EstimatedLossAmount = 3000, 
                    IncidentDescription = "Description", 
                    IncidentDate = DateTime.UtcNow, 
                    Status = "ClaimFiled",
                    IssuedPolicy = issuedPolicy
                }
            };

            _mockClaimService
                .Setup(s => s.GetPendingClaimsQueueAsync())
                .ReturnsAsync(mockPendingList);

            var result = await _controller.GetPendingQueue();

            var okResult = Assert.IsType<OkObjectResult>(result);
            var returnedList = Assert.IsAssignableFrom<IEnumerable<object>>(okResult.Value);
            Assert.Single(returnedList);
        }

        #endregion

        #region ReviewClaim Tests

        [Fact]
        public async Task ReviewClaim_ShouldReturnNotFound_WhenServiceReturnsFalse()
        {
            SetupUserIdentity("5", "Officer");
            int claimId = 1;
            var dto = new ClaimReviewDto { Status = "Approved", OfficerRemarks = "Approved" };

            _mockClaimService
                .Setup(s => s.ReviewClaimAsync(claimId, 5, dto))
                .ReturnsAsync(false);

            var result = await _controller.ReviewClaim(claimId, dto);

            Assert.IsType<NotFoundObjectResult>(result);
        }

        [Fact]
        public async Task ReviewClaim_ShouldReturnBadRequest_WhenServiceThrowsInvalidOperationException()
        {
            SetupUserIdentity("5", "Officer");
            int claimId = 1;
            var dto = new ClaimReviewDto { Status = "Approved", OfficerRemarks = "Approved" };

            _mockClaimService
                .Setup(s => s.ReviewClaimAsync(claimId, 5, dto))
                .ThrowsAsync(new InvalidOperationException("This claim has already been finalized."));

            var result = await _controller.ReviewClaim(claimId, dto);

            var badRequestResult = Assert.IsType<BadRequestObjectResult>(result);
            Assert.NotNull(badRequestResult.Value);
        }

        [Fact]
        public async Task ReviewClaim_ShouldReturnOk_WhenReviewSucceeds()
        {
            SetupUserIdentity("5", "Officer");
            int claimId = 1;
            var dto = new ClaimReviewDto { Status = "Approved", OfficerRemarks = "Approved" };

            _mockClaimService
                .Setup(s => s.ReviewClaimAsync(claimId, 5, dto))
                .ReturnsAsync(true);

            var result = await _controller.ReviewClaim(claimId, dto);

            var okResult = Assert.IsType<OkObjectResult>(result);
            Assert.NotNull(okResult.Value);
        }

        #endregion

        #region TrackClaimStatus Tests

        [Fact]
        public async Task TrackClaimStatus_ShouldReturnNotFound_WhenClaimIsNull()
        {
            SetupUserIdentity("123", "User");
            int claimId = 999;
            _mockClaimService
                .Setup(s => s.TrackClaimStatusAsync(claimId, 123, false))
                .ReturnsAsync((Claim?)null);

            var result = await _controller.TrackClaimStatus(claimId);

            Assert.IsType<NotFoundObjectResult>(result);
        }

        [Fact]
        public async Task TrackClaimStatus_ShouldReturnForbid_WhenServiceThrowsUnauthorizedAccessException()
        {
            SetupUserIdentity("123", "User");
            int claimId = 101;
            _mockClaimService
                .Setup(s => s.TrackClaimStatusAsync(claimId, 123, false))
                .ThrowsAsync(new UnauthorizedAccessException());

            var result = await _controller.TrackClaimStatus(claimId);

            Assert.IsType<ForbidResult>(result);
        }

        [Fact]
        public async Task TrackClaimStatus_ShouldReturnOk_WhenAuthorized()
        {
            SetupUserIdentity("123", "User");
            int claimId = 101;
            var claim = new Claim
            {
                ClaimId = claimId,
                IssuedPolicyId = 10,
                Status = "Approved",
                EstimatedLossAmount = 5000,
                ApprovedSettlementAmount = 4500,
                OfficerRemarks = "Approved",
                FiledAt = DateTime.UtcNow,
                UpdatedAt = DateTime.UtcNow
            };

            _mockClaimService
                .Setup(s => s.TrackClaimStatusAsync(claimId, 123, false))
                .ReturnsAsync(claim);

            var result = await _controller.TrackClaimStatus(claimId);

            var okResult = Assert.IsType<OkObjectResult>(result);
            Assert.NotNull(okResult.Value);
        }

        #endregion
    }
